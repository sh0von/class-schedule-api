API CURL Commands

Departments

curl -X GET http://localhost:3000/departments
curl -X POST -H "Content-Type: application/json" -d '{"name":"New Department"}' http://localhost:3000/departments
curl -X PUT -H "Content-Type: application/json" -d '{"name":"Updated Department"}' http://localhost:3000/departments/123
curl -X DELETE http://localhost:3000/departments/123

Courses

curl -X GET "http://localhost:3000/courses/?day=day&time=time&instructorId=instructor_id&departmentId=department_id&batchId=batch_id"
curl -X POST -H "Content-Type: application/json" -d '{"name":"New Course","instructor":"Instructor ID","department":"Department ID","batch":"Batch ID","time":"Course Time","day":"Course Day"}' http://localhost:3000/courses
curl -X PUT -H "Content-Type: application/json" -d '{"name":"Updated Course","instructor":"Instructor ID","department":"Department ID","batch":"Batch ID","time":"Course Time","day":"Course Day"}' http://localhost:3000/courses/123
curl -X DELETE http://localhost:3000/courses/123
